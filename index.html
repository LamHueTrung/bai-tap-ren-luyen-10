#include < stdio.h >
int sum(int a[], int st, int len);
int sum(int a[], int st, int len) {
	if (st >= len) {
		return 0;
	}
	return a[st] + sum(a, st + 1, len);
}
int main() { 
	int n;
	printf("kich thuoc mang ");
	scanf("%d", &n);
	int a[n];
	printf("\n nhap mang: \n ");
	for (int i = 0; i < n; i++) {
		scanf("%d", &a[i]);
	}
	printf("tong cua cac phan tu: %d", sum(a, 0, n));
	return 0;
}
